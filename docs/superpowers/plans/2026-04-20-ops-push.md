**Goal:** Ship hub-side catalog to OnPrintShop stores — `setProduct` + `setProductPrice` + `setProductCategory` + `setProductSize` + image upload — driven by an n8n workflow that FastAPI prepares payloads for, with per-customer markup applied and push history logged.

**Architecture:** Four independently shippable phases. Phase A extends the in-tree `n8n-nodes-onprintshop` TS node with the 4 critical write mutations. Phase B adds FastAPI push-orchestration (candidates + log + image processing). Phase C is the n8n push workflow. Phase D adds a "Publish to Store" UI on the admin catalog. Each phase produces working software on its own.

**Tech Stack:** n8n custom node (TypeScript, pnpm, tsc), FastAPI async SQLAlchemy, PostgreSQL, Pillow (image pipeline already exists), n8n workflow JSON, Next.js for UI.

**Scope note (plan bundles 4 subsystems):** Large; normally four specs. Kept together because the four phases share data contracts (`PushPayload`) and ordering — splitting created duplicate context. Execute phase by phase; each phase has a shippable acceptance gate.

```python
# backend/modules/push_log/routes.py
from typing import Optional
from uuid import UUID

from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from database import get_db

from .models import ProductPushLog
from .schemas import ProductPushLogCreate, ProductPushLogRead

router = APIRouter(prefix="/api/push-log", tags=["push_log"])


@router.get("", response_model=list[ProductPushLogRead])
async def list_push_log(
    product_id: Optional[UUID] = None,
    customer_id: Optional[UUID] = None,
    limit: int = 50,
    db: AsyncSession = Depends(get_db),
):
    q = select(ProductPushLog).order_by(ProductPushLog.pushed_at.desc()).limit(limit)
    if product_id:
        q = q.where(ProductPushLog.product_id == product_id)
    if customer_id:
        q = q.where(ProductPushLog.customer_id == customer_id)
    rows = (await db.execute(q)).scalars().all()
    return rows


@router.post("", response_model=ProductPushLogRead, status_code=201)
async def create_push_log(body: ProductPushLogCreate, db: AsyncSession = Depends(get_db)):
    entry = ProductPushLog(**body.model_dump())
    db.add(entry)
    await db.commit()
    await db.refresh(entry)
    return entry
```

```python
# backend/modules/push_candidates/service.py
from uuid import UUID

from sqlalchemy import and_, select
from sqlalchemy.ext.asyncio import AsyncSession

from modules.catalog.models import Product
from modules.customers.models import Customer
from modules.push_log.models import ProductPushLog


async def list_candidates(
    db: AsyncSession,
    customer_id: UUID,
    supplier_id: UUID | None = None,
    only_never_pushed: bool = False,
    limit: int = 100,
) -> list[Product]:
    """Products eligible to push to customer's OPS store.

    Criteria:
      - Product supplier_id matches argument (if provided)
      - Customer row is active
      - Product last_synced is not null (we have catalog data for it)
      - only_never_pushed=True filters out any product with an existing pushed
        log row for this customer.
    """
    customer = await db.get(Customer, customer_id)
    if not customer or not customer.is_active:
        return []

    q = select(Product).where(Product.last_synced.is_not(None))
    if supplier_id:
        q = q.where(Product.supplier_id == supplier_id)
    q = q.order_by(Product.last_synced.desc()).limit(limit)

    rows = (await db.execute(q)).scalars().all()

    if only_never_pushed:
        pushed_product_ids = {
            r[0]
            for r in (
                await db.execute(
                    select(ProductPushLog.product_id).where(
                        and_(
                            ProductPushLog.customer_id == customer_id,
                            ProductPushLog.status == "pushed",
                        )
                    )
                )
            ).all()
        }
        rows = [p for p in rows if p.id not in pushed_product_ids]

    return rows
```

```python
from .schemas import OPSProductInput  # near existing schema imports


def build_ops_input(payload: PushPayload) -> OPSProductInput:
    """Flatten PushPayload into OPS setProduct input shape.

    OPS keys default_category_id/brand_id by integer — pass-through only
    when the source carries them (currently null for VG OPS pull; SanMar
    suppliers will populate).
    """
    first_variant = payload.variants[0] if payload.variants else None
    base_price = first_variant.final_price if first_variant else None
    return OPSProductInput(
        products_title=payload.meta.product_name,
        products_sku=payload.meta.supplier_sku,
        products_description=payload.meta.description,
        status=1,
        default_category_id=int(payload.meta.ops_category_id) if payload.meta.ops_category_id else None,
        brand_id=None,
        products_price=str(base_price) if base_price is not None else None,
        small_image=payload.meta.primary_image_url,
        large_image=payload.meta.primary_image_url,
    )
```

```tsx
"use client";

import { useEffect, useState } from "react";
import { api } from "@/lib/api";
import type { Customer } from "@/lib/types";

interface Props {
  productId: string;
  onDone?: () => void;
}

const N8N_URL = process.env.NEXT_PUBLIC_N8N_URL ?? "http://localhost:5678";

export function PublishButton({ productId, onDone }: Props) {
  const [customers, setCustomers] = useState<Customer[]>([]);
  const [customerId, setCustomerId] = useState<string>("");
  const [busy, setBusy] = useState(false);
  const [message, setMessage] = useState<string | null>(null);

  useEffect(() => {
    api<Customer[]>("/api/customers").then((list) => {
      setCustomers(list);
      const first = list.find((c) => c.is_active);
      if (first) setCustomerId(first.id);
    });
  }, []);

  async function run() {
    if (!customerId) {
      setMessage("Pick a customer first");
      return;
    }
    setBusy(true);
    setMessage(null);
    try {
      // Fires the single-product push path on the backend, which in turn
      // triggers the n8n push workflow for this product + customer pair.
      const res = await api<{ triggered: boolean }>(
        `/api/n8n/workflows/vg-ops-push-001/trigger?product_id=${productId}&customer_id=${customerId}`,
        { method: "POST" },
      );
      if (res.triggered) {
        setMessage("Push started. Watch history below.");
        onDone?.();
      } else {
        setMessage("Push request failed.");
      }
    } catch (err) {
      setMessage(err instanceof Error ? err.message : String(err));
    } finally {
      setBusy(false);
    }
  }

  return (
    <div className="flex items-center gap-3">
      <select
        value={customerId}
        onChange={(e) => setCustomerId(e.target.value)}

      >
        <option value="">Select customer…</option>
        {customers.map((c) => (
          <option key={c.id} value={c.id} disabled={!c.is_active}>
            {c.name} {c.is_active ? "" : "(inactive)"}
          </option>
        ))}
      </select>
      <button
        type="button"
        onClick={run}
        disabled={busy || !customerId}
        className="px-4 py-2 rounded-md bg-[#1e4d92] text-white text-[12px] font-semibold hover:bg-[#163f78] disabled:opacity-50"
      >
        {busy ? "Pushing…" : "Publish to OPS"}
      </button>
      {message && (
        <div className="text-[11px] font-mono text-[#484852]">{message}</div>
      )}
      <a
        href={`${N8N_URL}/workflow/vg-ops-push-001`}
        target="_blank" rel="noopener noreferrer"
        className="text-[11px] font-semibold text-[#484852] hover:text-[#1e4d92]"
      >
        Open workflow ↗
      </a>
    </div>
  );
}
```
